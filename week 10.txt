class Node: def init(self, content): self.content = content self.next = None

class LinkedList: def init(self): self.front = None self.rear = None

def add_front(self, new): new_node = Node(new) if self.front is None: self.front = new_node self.rear = new_node else: new_node.next = self.front self.front = new_node

def add_rear(self, new): new_node = Node(new) if self.rear is None: self.front = new_node self.rear = new_node else: self.rear.next = new_node self.rear = new_node

def print_list(self): tmp = self.front while tmp is not None: print(tmp.content) tmp = tmp.next my_str = "Today I studied linked lists in my class." words = my_str.split(' ')

ll = LinkedList()

for word in words: print(len(word)) ll.add_rear(len(word))

print("\nLinked list içeriği:") ll.print_list()
